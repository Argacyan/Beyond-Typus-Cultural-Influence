province_event = { #Serbo-Croat settlements and assimilation (Settle)
	id = cultural_influence_south_slavic.4
	title = serbo_croat_settlements.title
	desc = serbo_croat_settlements.desc
	picture = CULTURE_eventPicture

	trigger = {
		owner = { culture_group = south_slavic }
		NOT = { culture = serbian }
		NOT = { culture = croatian }
		NOT = { culture = bosnian }
		OR = { #Permit AI to settle accepted cultures since they don't understand
			has_owner_accepted_culture = no
			owner = { ai = yes }
		}
		OR = { #Intent check for beings capable of signaling intent (Requiring this allows lower MTTHs and hopefully more directed efforts)
			AND = { #AIs are dumb (With a lot of work it would be possible to use a system of decisions and flags to signal intent here)
				owner = { ai = yes }
				any_neighbor_province = { culture_group = south_slavic }
			}
			num_of_times_improved_by_owner = 5 #Non-Dharma intent check
			AND = { #Dharma intent check
				has_colonist = yes
				is_city = yes
			}
		}
		owner = { has_country_modifier = south_slavic_settlements }
		serbo_croat_culture_region = yes
		has_global_flag = CI_events_enabled
	}

	mean_time_to_happen = {
		#Template Settled MTTH
		years = 10

		modifier = {
			factor = 0.5
			has_global_flag = 05_CI_MTTH_flag
		}
		modifier = {
			factor = 1.5
			has_global_flag = 15_CI_MTTH_flag
		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}
		modifier = {
			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}

		modifier = { #Non-Dharma
			factor = 3
			OR = { #ai uses thiS one
				NOT = { has_dlc = "Dharma" }
				owner = { ai = yes }
			}
		}
		modifier = {
			factor = 0.8
			num_of_times_improved_by_owner = 5
			NOT = { has_dlc = "Dharma" }
		}
		modifier = {
			factor = 0.6
			num_of_times_improved_by_owner = 10
			NOT = { has_dlc = "Dharma" }
		}
		modifier = {
			factor = 0.8
			has_idea_to_settle_faster = yes
		}
		modifier = {
			factor = 1.4
			NOT = { development = 10 }
		}
		modifier = {
			factor = 1.2
			NOT = { development = 20 }
		}
		#Ideal dev is around 30
		modifier = {
			factor = 1.2
			development = 40
		}
		modifier = {
			factor = 1.4
			development = 50
		}
		modifier = {
			factor = 1.2
			has_owner_religion = no
		}
		modifier = {
			factor = 0.9
			adm = 4
		}
		modifier = {
			factor = 0.9
			adm = 5
		}
		modifier = {
			factor = 0.9
			adm = 6
		}
		modifier = {
			factor = 0.75
			national_focus = adm
		}
	}

	option = {
		name = serbo_croat_settlements.a

		if = { #Using religion of the province to determine what dialect to use
			limit = {
				religion = orthodox
			}
			change_culture = serbian
		}
		if = {
			limit = {
				religion_group = christian
				NOT = { religion = orthodox }
			}
			change_culture = croatian
		}
		if = {
			limit = {
				religion_group = muslim
			}
			change_culture = bosnian
		}
		if = { #Western Shtokavian owner
			limit = {
				NOT = { religion_group = christian }
				NOT = { religion_group = muslim }
				OR = {
					owner = { primary_culture = slovene }
					owner = { primary_culture = croatian }
					owner = { primary_culture = bosnian }
				}
			}
			change_culture = croatian
		}
		if = {
			limit = {
				NOT = { religion_group = christian }
				NOT = { religion_group = muslim }
				OR = {
					owner = { primary_culture = serbian }
					owner = { primary_culture = bulgarian }
				}
			}
			change_culture = serbian
		}
	}
}

	desc = serbo_croat_influences.desc
	picture = RELIGIOUS_CONVERSION_eventPicture

	trigger = {
		OR = {
			OWNER = { primary_culture = croatian }
			OWNER = { primary_culture = bosnian }
			OWNER = { primary_culture = serbian }
		}
		NOT = { culture = croatian }
		NOT = { culture = bosnian }
		NOT = { culture = serbian }
		culture_group = south_slavic
		OR = {
			any_neighbor_province = { culture = croatian }
			any_neighbor_province = { culture = bosnian }

		}

		}
		modifier = {
			factor = 2.0
			has_global_flag = 20_CI_MTTH_flag
		}

			factor = 4.0
			has_global_flag = 40_CI_MTTH_flag
		}

		#Neighbours
		modifier = {
			factor = 0.75
			calc_true_if = {
				all_neighbor_province = {
					OR = {
						culture = croatian
						culture = bosnian
						culture = serbian

			factor = 0.75
			national_focus = adm
		}
