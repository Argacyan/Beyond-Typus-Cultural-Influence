country_event = { #German Standardization
	id = cultural_influence_germanic.1
	title = german_standardization.title
	desc = german_standardization.desc

	trigger = {

			primary_culture = prussian
			primary_culture = frisian
			primary_culture = swiss
		}
		NOT = { has_global_flag = german_standardized }
		NOT = {
			any_province = {
				AND = {
					hochdeutsch_culture_group = yes
					NOT = { owned_by = ROOT }
					continent = europe
				}
			}
		}
		has_global_flag = CI_events_enabled
	}

		name = german_standardization.a

		trigger = {
			owns = 134
		}

		custom_tooltip = german_standardization.effect
		hidden_effect = {
			set_global_flag = german_standardized
			change_primary_culture = standard_german
			set_ruler_culture = standard_german
			set_heir_culture = standard_german
			if = {
				limit = {
					NOT = { has_foreign_consort = yes }
				}
				set_consort_culture = standard_german
			}
			every_province = {
				limit = {
					OR = {
						culture = austrian
						hochdeutsch_culture_group = yes
						AND = {
							culture = austrian
							ROOT = { primary_culture = austrian }
						}
						AND = {
							culture = frisian
							ROOT = { primary_culture = frisian }
						}
						AND = {
							culture = swiss
							ROOT = { primary_culture = swiss }
						}

province_event = { #Austrian integration
	id = cultural_influence_germanic.3
	title = austrian_integration.title
	desc = austrian_integration.desc
	picture = CULTURE_eventPicture

	trigger = {
		has_global_flag = german_standardized
		culture = austrian
		owner = {
			government_rank = 3
			primary_culture = standard_german
		}
		OR = { #Lack of standard German influence or signifigant transit
			development = 20 
			province_trade_power = 10
			any_neighbor_province = {
				culture = standard_german
			}
		}
		has_global_flag = CI_events_enabled
	}

	mean_time_to_happen = {
		years = 5 #No historic resistance, just not part of Germany
		modifier = { #Rural communities reluctant to convert
			factor = 1.2
			NOT = { development = 10 }
		}
		modifier = { #More extreme rural communities
			factor = 1.5
			NOT = { development = 5 }
		}
		modifier = { #Townships more connected to capital
			factor = 0.9
			development = 10
		}
		modifier = { #Cities even more connected
			factor = 0.8
			development = 20
		}
		modifier = { #Larger city means more connection
			factor = 0.7
			development = 30
		}
		modifier = { #Stop developing pls. k thx
			factor = 0.6
			development = 40
		}
	}

	option = {
		name = austrian_integration.a

		change_culture = standard_german
	}
}

		desc = swiss_assimilation_swiss.desc

			any_neighbor_province = { culture = swiss }
		}
	}
	desc = {
		desc = swiss_assimilation_hochdeutsch.desc
		trigger = {
			NOT = { any_neighbor_province = { culture = swiss } }
		}
	}
	picture = STREET_SPEECH_eventPicture

	trigger = {
		always = no #Disabled

			area = switzerland_area

			OR = {
				primary_culture = standard_german
				primary_culture = swiss
			}
		}
		any_neighbor_province = {
			OR = {

				culture = swiss
			}
		}
	}

	mean_time_to_happen = {
		years = 50

		}
		modifier = { #Cities even more connected
			factor = 0.8

		}
		modifier = { #Larger city means more connection
			factor = 0.7
			development = 30
		}
		modifier = { #Stop developing pls. k thx
			factor = 0.6
			development = 40
		}
	}

	option = {
		name = swiss_assimilation.a
